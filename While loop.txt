public class WhileLoopExample {
    public static void main(String[] args) {
        int i = 1;

        System.out.println("Printing numbers from 1 to 10 using a while loop:");

        // while loop to print numbers 1 to 10
        while (i <= 10) {
            System.out.println(i);
            i++;
        }
    }
}
