public class DoWhileExample {
    public static void main(String[] args) {
        int i = 1;

        System.out.println("Printing numbers from 1 to 10 using do-while loop:");

        // do-while loop to print numbers from 1 to 10
        do {
            System.out.println(i);
            i++;
        } while (i <= 10);
    }
}
